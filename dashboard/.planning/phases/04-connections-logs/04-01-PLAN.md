```typescript
export async function closeConnection(id: string): Promise<void> {
  const res = await fetch(`${getBaseUrl()}/connections/${encodeURIComponent(id)}`, {
    method: 'DELETE',
    headers: getHeaders(),
    signal: AbortSignal.timeout(5000),
  })
  if (!res.ok && res.status !== 204) {
    throw new Error('Failed to close connection')
  }
}

export async function closeAllConnections(): Promise<void> {
  const res = await fetch(`${getBaseUrl()}/connections`, {
    method: 'DELETE',
    headers: getHeaders(),
    signal: AbortSignal.timeout(5000),
  })
  if (!res.ok && res.status !== 204) {
    throw new Error('Failed to close all connections')
  }
}
```

**Step 3: Create connections store** (`src/stores/connections.ts`)

Volatile Zustand store (no persist), following proxies.ts pattern:

```typescript
interface ConnectionWithSpeed extends Connection {
  dlSpeed: number  // bytes/sec calculated from snapshot delta
  ulSpeed: number
}

interface ConnectionsState {
  // Data
  connections: ConnectionWithSpeed[]
  totalUpload: number
  totalDownload: number

  // UI state
  paused: boolean
  searchQuery: string
  networkFilter: string      // 'all' | 'tcp' | 'udp'
  ruleFilter: string         // 'all' | specific rule
  chainFilter: string        // 'all' | specific proxy chain
  expandedId: string | null
  visibleColumns: string[]

  // Internal (for speed calc)
  _prevSnapshot: Map<string, { upload: number; download: number }> | null
  _prevTime: number

  // Computed
  filteredConnections: () => ConnectionWithSpeed[]
  uniqueRules: () => string[]
  uniqueChains: () => string[]

  // Actions
  updateSnapshot: (snapshot: ConnectionsSnapshot) => void
  setPaused: (p: boolean) => void
  setSearchQuery: (q: string) => void
  setNetworkFilter: (f: string) => void
  setRuleFilter: (f: string) => void
  setChainFilter: (f: string) => void
  setVisibleColumns: (cols: string[]) => void
  toggleExpanded: (id: string) => void
  closeConnection: (id: string) => Promise<void>
  closeAllConnections: () => Promise<void>
}
```

Default visible columns: `['host', 'network', 'source', 'destination', 'rule', 'chains', 'dlSpeed', 'ulSpeed']`

Store `visibleColumns` in localStorage manually (like expandedGroups in proxies.ts) so column preferences persist across sessions.

The `updateSnapshot` action:
- If `paused`, return immediately (don't update)
- Calculate speed delta: for each connection in new snapshot, compare upload/download with `_prevSnapshot` map by connection id. Speed = (currentBytes - prevBytes) / timeDeltaSeconds. New connections get 0 speed.
- Store current snapshot as `_prevSnapshot` for next delta
- Set `connections` to the new array with speeds

The `filteredConnections` getter function:
- Filter by networkFilter (metadata.network === 'tcp'/'udp')
- Filter by ruleFilter (connection.rule)
- Filter by chainFilter (connection.chains[0] or joined chains)
- Filter by searchQuery: case-insensitive match against host, sourceIP, destinationIP, rule, chains joined, processPath
- Return filtered array

The `closeConnection` action: call `closeConnection(id)` from mihomo-api, toast on error. The connection will disappear from next snapshot automatically.

The `closeAllConnections` action: show AlertDialog confirmation (handled in UI), then call `closeAllConnections()` from mihomo-api, toast on success/error.

**Step 4: Create logs store** (`src/stores/logs.ts`)

```typescript
const MAX_LOG_ENTRIES = 1000

interface LogEntry {
  id: number
  time: string       // HH:MM:SS from structured format
  level: string      // 'debug' | 'info' | 'warning' | 'error'
  message: string
  fields: Array<{ key: string; value: string }>
}

interface LogsState {
  entries: LogEntry[]
  nextId: number
  activeLevels: Set<string>   // Which levels shown (all active by default)
  searchQuery: string
  paused: boolean

  // Computed
  filteredEntries: () => LogEntry[]

  // Actions
  addEntry: (raw: { time: string; level: string; message: string; fields?: Array<{ key: string; value: string }> }) => void
  clear: () => void
  toggleLevel: (level: string) => void
  setSearchQuery: (q: string) => void
  setPaused: (p: boolean) => void
  exportTxt: () => void
  exportJson: () => void
}
```

The `addEntry` action:
- If paused, still add to buffer (don't lose data), but UI won't auto-scroll (handled by component)
- Normalize level: map 'warn' -> 'warning' (mihomo sends 'warn' in structured format)
- Create LogEntry with auto-incrementing id
- Append to entries array. If entries.length > MAX_LOG_ENTRIES, slice to keep only last MAX_LOG_ENTRIES (ring buffer via shift)

The `filteredEntries` getter:
- Filter by activeLevels (entry.level must be in set)
- Filter by searchQuery: case-insensitive match on message and fields values

The `exportTxt` and `exportJson` actions:
- Use native Blob + URL.createObjectURL + anchor download pattern (as shown in research)
- Export ALL entries (not just filtered), with filename `mihomo-logs-{timestamp}.txt` / `.json`
- TXT format: `[HH:MM:SS] [LEVEL] message`
- JSON format: pretty-printed array of LogEntry objects

The `clear` action: reset entries to [], nextId to 0.

**Step 2: Create ConnectionsLogsPage.tsx**

This is the single page with two tabs per user decision. Uses shadcn Tabs component.

```tsx
import { useCallback } from 'react'
import { Tabs, TabsContent, TabsList, TabsTrigger } from '@/components/ui/tabs'
import { useMihomoWs } from '@/hooks/use-mihomo-ws'
import { useConnectionsStore } from '@/stores/connections'
import { useLogsStore } from '@/stores/logs'
import { ConnectionsTab } from '@/components/connections/ConnectionsTab'
import { LogsTab } from '@/components/logs/LogsTab'
import type { ConnectionsSnapshot } from '@/lib/mihomo-api'

interface LogStructuredMessage {
  time: string
  level: string
  message: string
  fields?: Array<{ key: string; value: string }>
}

export default function ConnectionsLogsPage() {
  const updateSnapshot = useConnectionsStore((s) => s.updateSnapshot)
  const addLogEntry = useLogsStore((s) => s.addEntry)

  const handleConnections = useCallback(
    (data: ConnectionsSnapshot) => updateSnapshot(data),
    [updateSnapshot]
  )

  const handleLog = useCallback(
    (data: LogStructuredMessage) => addLogEntry(data),
    [addLogEntry]
  )

  // WebSocket streams -- always active regardless of which tab is shown
  useMihomoWs<ConnectionsSnapshot>('/connections', handleConnections, 1000)
  useMihomoWs<LogStructuredMessage>('/logs?level=debug&format=structured', handleLog)

  return (
    <Tabs defaultValue="connections" className="flex flex-col h-full">
      <TabsList className="w-fit">
        <TabsTrigger value="connections">Подключения</TabsTrigger>
        <TabsTrigger value="logs">Логи</TabsTrigger>
      </TabsList>

      <TabsContent value="connections" className="flex-1 mt-2 min-h-0 data-[state=active]:animate-in data-[state=active]:fade-in-0 data-[state=active]:duration-200">
        <ConnectionsTab />
      </TabsContent>

      <TabsContent value="logs" className="flex-1 mt-2 min-h-0 data-[state=active]:animate-in data-[state=active]:fade-in-0 data-[state=active]:duration-200">
        <LogsTab />
      </TabsContent>
    </Tabs>
  )
}
```

Key points:
- WebSocket hooks run at the page level so BOTH streams are always active regardless of active tab
- `useMihomoWs` for `/connections` with interval=1000 (1 second updates)
- `useMihomoWs` for `/logs` with query params `level=debug&format=structured`. Note: the useMihomoWs hook appends `?token=SECRET` -- check how it handles paths with existing query params. If the hook uses URLSearchParams.set(), additional params in the path may need special handling. The safest approach: pass `/logs` as path, and modify useMihomoWs to accept additional query params, OR just include level and format as part of the path string since the hook appends `?token=X&interval=Y` with URLSearchParams which would overwrite. **Actually**, looking at the hook code -- it constructs `params` and builds `fullUrl = query ? wsUrl + path + ? + query : wsUrl + path`. If path already contains `?`, this breaks. Two solutions: (a) modify the hook to accept extra params, or (b) set the path to `/logs` and add `level` and `format` as new optional params. **Choose option (b)**: extend `useMihomoWs` to accept an optional `params?: Record<string, string>` argument. Update the hook to merge these with token/interval params. This is backward-compatible.
- Tab animation: `data-[state=active]:animate-in data-[state=active]:fade-in-0 data-[state=active]:duration-200`
